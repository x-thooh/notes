```go
func merge(list []uint32) []uint32 {
	if len(list) <= 1 {
		return list
	}
	m := len(list) / 2
	a := merge(list[:m])
	b := merge(list[m:])
	i, j := 0, 0
	c := make([]uint32, 0, len(a)+len(b))
	for i < len(a) && j < len(b) {
		if a[i] < b[j] {
			c = append(c, a[i])
			i++
		} else {
			c = append(c, b[j])
			j++
		}
	}
	if i < len(a) {
		c = append(c, a[i:]...)
	} else {
		c = append(c, b[j:]...)
	}
	return c
}
```
