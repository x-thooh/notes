# 二叉树

## 二叉树遍历

### 深度优先遍历（DFS）

前、中、后序遍历都属于**深度优先遍历**。它们的命名规律非常简单：**“前、中、后”指的是【根节点】被访问的顺序**。无论哪种，左子树永远先于右子树被访问（$L \rightarrow R$）。

#### 树的结构示例
假设我们有一棵简单的二叉树：
```shell
    A (根)
   / \
  B   C (右)
 /
D (左)
```

#### 轴向逻辑拆解

- ### **前序遍历 (Pre-order)**
    
    - **顺序**：**根** $\rightarrow$ 左 $\rightarrow$ 右 （N $\rightarrow$ L $\rightarrow$ R）
        
    - **记忆点**：最先访问根节点。从上往下，先探顶，再深入。
        
    - **上述树的遍历结果**：`A -> B -> D -> C`
        
    - **应用场景**：常用于复制一棵树、生成树的结构化目录。
        
- ### **中序遍历 (In-order)**
    
    - **顺序**：左 $\rightarrow$ **根** $\rightarrow$ 右 （L $\rightarrow$ N $\rightarrow$ R）
        
    - **记忆点**：根节点在中间访问。从左下角开始，像一条水平线从左往右扫描。
        
    - **上述树的遍历结果**：`D -> B -> A -> C`
        
    - **应用场景**：在**二叉搜索树（BST）**中，中序遍历的结果是一个**严格升序**的有序序列。
        
- ### **后序遍历 (Post-order)**
    
    - **顺序**：左 $\rightarrow$ 右 $\rightarrow$ **根** （L $\rightarrow$ R $\rightarrow$ N）
        
    - **记忆点**：最后访问根节点。先剪除两边的叶子，最后回到总根。
        
    - **上述树的遍历结果**：`D -> B -> C -> A`
        
    - **应用场景**：常用于**计算文件/目录的大小**、删除整棵树（必须先删子节点，才能删父节点）以及表达式树的逆波兰式求解。
        

### 广度优先遍历（BFS）

- ### **层级遍历 (Level-order)**
    
    - **顺序**：从上到下，从左到右，一层一层地访问。
        
    - **核心逻辑**：先访问完第 $h$ 层的所有节点，再进入第 $h+1$ 层。
        
    - **上述树的遍历结果**：`A -> B -> C -> D`（第一层 A，第二层 B、C，第三层 D）
        
    - **应用场景**：寻找二叉树中的最短路径、寻找某节点的左右邻居（兄弟节点）。
        
    - **实现工具**：代码实现时，深度优先通常用**递归**或**栈（Stack）**；而层级遍历必须借助队列（Queue）来先进先出地暂存节点。
### 递归实现
```go
package main

import "fmt"

// TreeNode 定义二叉树节点
type TreeNode struct {
	Val   int
	Left  *TreeNode
	Right *TreeNode
}

// 1. 前序遍历 (Pre-order): 根 -> 左 -> 右
func preorderTraversal(root *TreeNode) []int {
	var res []int
	var dfs func(*TreeNode)
	dfs = func(node *TreeNode) {
		if node == nil {
			return
		}
		res = append(res, node.Val) // 根
		dfs(node.Left)              // 左
		dfs(node.Right)             // 右
	}
	dfs(root)
	return res
}

// 2. 中序遍历 (In-order): 左 -> 根 -> 右
func inorderTraversal(root *TreeNode) []int {
	var res []int
	var dfs func(*TreeNode)
	dfs = func(node *TreeNode) {
		if node == nil {
			return
		}
		dfs(node.Left)              // 左
		res = append(res, node.Val) // 根
		dfs(node.Right)             // 右
	}
	dfs(root)
	return res
}

// 3. 后序遍历 (Post-order): 左 -> 右 -> 根
func postorderTraversal(root *TreeNode) []int {
	var res []int
	var dfs func(*TreeNode)
	dfs = func(node *TreeNode) {
		if node == nil {
			return
		}
		dfs(node.Left)              // 左
		dfs(node.Right)             // 右
		res = append(res, node.Val) // 根
	}
	dfs(root)
	return res
}

// 4. 层级遍历 (Level-order): 广度优先，借助队列
func levelOrder(root *TreeNode) [][]int {
	var res [][]int
	if root == nil {
		return res
	}

	// 初始化队列，放入根节点
	queue := []*TreeNode{root}

	for len(queue) > 0 {
		levelSize := len(queue) // 当前层的节点数量
		var currentLevel []int

		// 遍历当前层的所有节点
		for i := 0; i < levelSize; i++ {
			node := queue[0] // 出队
			queue = queue[1:]

			currentLevel = append(currentLevel, node.Val)

			// 将下一层的子节点推入队列
			if node.Left != nil {
				queue = append(queue, node.Left)
			}
			if node.Right != nil {
				queue = append(queue, node.Right)
			}
		}
		// 将当前层的结果整行加入最终结果
		res = append(res, currentLevel)
	}
	return res
}

func main() {
	/*
	    构建测试二叉树:
	         1
	        / \
	       2   3
	      / \
	     4   5
	*/
	root := &TreeNode{Val: 1}
	root.Left = &TreeNode{Val: 2}
	root.Right = &TreeNode{Val: 3}
	root.Left.Left = &TreeNode{Val: 4}
	root.Left.Right = &TreeNode{Val: 5}

	fmt.Println("前序遍历:", preorderTraversal(root)) // [1 2 4 5 3]
	fmt.Println("中序遍历:", inorderTraversal(root))  // [4 2 5 1 3]
	fmt.Println("后序遍历:", postorderTraversal(root)) // [4 5 2 3 1]
	fmt.Println("层级遍历:", levelOrder(root))        // [[1] [2 3] [4 5]]
}
```

### 迭代实现
```go
package main

import "fmt"

type TreeNode struct {
	Val   int
	Left  *TreeNode
	Right *TreeNode
}

// 1. 前序遍历 (迭代): 根 -> 左 -> 右
// 核心思想：每次出栈一个节点，由于栈是后进先出(LIFO)，所以要先压入右子树，再压入左子树
func preorderTraversalIteration(root *TreeNode) []int {
	var res []int
	if root == nil {
		return res
	}

	stack := []*TreeNode{root}
	for len(stack) > 0 {
		// 出栈
		node := stack[len(stack)-1]
		stack = stack[:len(stack)-1]

		res = append(res, node.Val)

		// 先右后左入栈，这样出栈时就是先左后右
		if node.Right != nil {
			stack = append(stack, node.Right)
		}
		if node.Left != nil {
			stack = append(stack, node.Left)
		}
	}
	return res
}

// 2. 中序遍历 (迭代): 左 -> 根 -> 右
// 核心思想：一路向左推到底入栈，直到没有左子树了，再弹出节点（访问根），然后转向右子树
func inorderTraversalIteration(root *TreeNode) []int {
	var res []int
	var stack []*TreeNode
	curr := root

	for curr != nil || len(stack) > 0 {
		// 尽可能往左走，把左边界全部压栈
		for curr != nil {
			stack = append(stack, curr)
			curr = curr.Left
		}

		// 弹出栈顶节点并访问
		curr = stack[len(stack)-1]
		stack = stack[:len(stack)-1]
		res = append(res, curr.Val)

		// 转向右子树
		curr = curr.Right
	}
	return res
}

// 3. 后序遍历 (迭代): 左 -> 右 -> 根
// 核心思想（双栈法简化/逆序法）：前序是 根->左->右，如果改成 根->右->左，
// 那么把它整体翻转过来，就是 左->右->根（正好是后序）。
func postorderTraversalIteration(root *TreeNode) []int {
	var res []int
	if root == nil {
		return res
	}

	stack := []*TreeNode{root}
	for len(stack) > 0 {
		node := stack[len(stack)-1]
		stack = stack[:len(stack)-1]

		res = append(res, node.Val)

		// 先左后右入栈，出栈顺序就是先右后左
		if node.Left != nil {
			stack = append(stack, node.Left)
		}
		if node.Right != nil {
			stack = append(stack, node.Right)
		}
	}

	// 翻转结果切片
	for i, j := 0, len(res)-1; i < j; i, j = i+1, j-1 {
		res[i], res[j] = res[j], res[i]
	}
	return res
}

// 4. 层级遍历 (迭代): 借助队列（原生即为迭代）
func levelOrderIteration(root *TreeNode) [][]int {
	var res [][]int
	if root == nil {
		return res
	}

	queue := []*TreeNode{root}
	for len(queue) > 0 {
		levelSize := len(queue)
		var currentLevel []int

		for i := 0; i < levelSize; i++ {
			node := queue[0]
			queue = queue[1:]

			currentLevel = append(currentLevel, node.Val)

			if node.Left != nil {
				queue = append(queue, node.Left)
			}
			if node.Right != nil {
				queue = append(queue, node.Right)
			}
		}
		res = append(res, currentLevel)
	}
	return res
}

func main() {
	// 构建测试二叉树
	//      1
	//     / \
	//    2   3
	//   / \
	//  4   5
	root := &TreeNode{Val: 1}
	root.Left = &TreeNode{Val: 2}
	root.Right = &TreeNode{Val: 3}
	root.Left.Left = &TreeNode{Val: 4}
	root.Left.Right = &TreeNode{Val: 5}

	fmt.Println("迭代前序:", preorderTraversalIteration(root)) // [1 2 4 5 3]
	fmt.Println("迭代中序:", inorderTraversalIteration(root))  // [4 2 5 1 3]
	fmt.Println("迭代后序:", postorderTraversalIteration(root)) // [4 5 2 3 1]
	fmt.Println("迭代层级:", levelOrderIteration(root))        // [[1] [2 3] [4 5]]
}
```
